🔍 Debug Admin Panel – PLN UP3 Jayapura

Buka halaman ini di tab YANG SAMA dengan aplikasi PLN (domain sama), atau copy-paste kode JS di bawah ke Console browser.

📋 Instruksi:
1. Login ke aplikasi PLN dulu di tab lain
2. Kembali ke tab ini, klik "Jalankan Debug"
3. Atau: buka DevTools (F12) → Console → paste kode JS di bawah
🔧 Kode Debug (paste ke Console): 
(function() {
  var out = [];

  // 1. Cek session
  try {
    var s = JSON.parse(sessionStorage.getItem('plnSession') || 'null');
    out.push('=== SESSION ===');
    out.push(s ? JSON.stringify(s, null, 2) : 'NULL - belum login!');
    if (s) {
      out.push('role: ' + s.role);
      out.push('role === superadmin: ' + (s.role === 'superadmin'));
    }
  } catch(e) { out.push('Session error: ' + e); }

  out.push('');
  out.push('=== MODAL ADMIN PANEL ===');

  // 2. Cek modal
  var modal = document.getElementById('modal-admin-panel');
  out.push('modal-admin-panel ada: ' + !!modal);
  if (modal) out.push('display: ' + modal.style.display);

  // 3. Cek tombol
  var btns = ['apbtn-tambah','apbtn-edit','apbtn-export','apbtn-user',
              'apbtn-usulp','apbtn-pin','apbtn-import','apbtn-logout'];
  out.push('');
  out.push('=== TOMBOL DI DOM ===');
  btns.forEach(function(id) {
    var el = document.getElementById(id);
    if (el) {
      out.push(id + ': ADA | display=' + (el.style.display||'(default)') + ' | text=' + el.textContent.trim().substring(0,30));
    } else {
      out.push(id + ': TIDAK ADA di DOM!');
    }
  });

  // 4. Cek apakah _tampilkanPanelAdmin ada
  out.push('');
  out.push('=== FUNGSI JS ===');
  out.push('_tampilkanPanelAdmin: ' + (typeof _tampilkanPanelAdmin));
  out.push('bukaMenuAdmin: ' + (typeof bukaMenuAdmin));
  out.push('bukaModalCsvImport: ' + (typeof bukaModalCsvImport));

  // 5. Cek admin-panel-actions innerHTML
  var actions = document.getElementById('admin-panel-actions');
  out.push('');
  out.push('=== admin-panel-actions ===');
  if (actions) {
    out.push('Ada: ya');
    out.push('Children count: ' + actions.children.length);
    for (var i = 0; i < actions.children.length; i++) {
      var c = actions.children[i];
      out.push('[' + i + '] id=' + (c.id||'(no-id)') + ' display=' + (c.style.display||'default') + ' text=' + c.textContent.trim().substring(0,35));
    }
  } else {
    out.push('TIDAK ADA admin-panel-actions!');
  }

  // 6. Simulate buka panel
  out.push('');
  out.push('=== SIMULASI bukaMenuAdmin() ===');
  try {
    if (typeof bukaMenuAdmin === 'function') {
      bukaMenuAdmin();
      out.push('bukaMenuAdmin() berhasil dipanggil');
      // Cek ulang tombol setelah panel dibuka
      var imp = document.getElementById('apbtn-import');
      if (imp) {
        out.push('apbtn-import setelah buka: display=' + (imp.style.display||'default'));
      }
    } else {
      out.push('bukaMenuAdmin tidak ditemukan!');
    }
  } catch(e) { out.push('Error: ' + e); }

  var result = out.join('\n');
  console.log(result);
  document.getElementById('debug-output').textContent = result;
  document.getElementById('debug-output').style.color = '#4ade80';
})();

🔧 Quick Fix — Paksa tampilkan tombol Import: 
(function(){
  // Tambahkan tombol import langsung ke panel
  var actions = document.getElementById('admin-panel-actions');
  if (!actions) { console.error('admin-panel-actions tidak ada!'); return; }
  
  // Cek apakah sudah ada
  if (document.getElementById('apbtn-import')) {
    document.getElementById('apbtn-import').style.display = '';
    console.log('apbtn-import sudah ada, display direset');
    return;
  }
  
  // Buat tombol baru
  var btn = document.createElement('button');
  btn.id = 'apbtn-import';
  btn.className = 'btn btn-outline';
  btn.textContent = '📥 Import Bulk Inspeksi';
  btn.onclick = function() {
    tutupPanelAdmin();
    bukaModalCsvImport();
  };
  
  // Sisipkan sebelum tombol Logout
  var logout = document.getElementById('apbtn-logout');
  if (logout) {
    actions.insertBefore(btn, logout);
  } else {
    actions.appendChild(btn);
  }
  console.log('Tombol Import Bulk berhasil ditambahkan!');
})();
